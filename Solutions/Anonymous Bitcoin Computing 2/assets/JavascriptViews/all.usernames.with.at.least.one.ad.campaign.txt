Reduce:

function(keys, values, rereduce)
{
  var ret = [];
  var retLen = 0;
  var previousKey = "_\__ _|__/ _  __";
  var keyLen = keys.length;
  for(i = 0; i < keyLen; i++)
  {
    var currentKey = keys[i];
    if(currentKey != previousKey)
    {
      ret[retLen] = currentKey;
      retLen++;
      previousKey = currentKey;
    }
  }
  return ret;
}
