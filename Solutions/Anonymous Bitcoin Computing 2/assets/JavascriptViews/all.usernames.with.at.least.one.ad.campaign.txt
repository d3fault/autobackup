Reduce:

function(keys, values, rereduce)
{
  var ret = [];
  var keyLen = keys.length;
  for(i = 0; i < keyLen; i++)
  {
    var currentKey = keys[i];
    var retLen = ret.length;
    var alreadyInRet = false;
    for(j = 0; j < retLen; j++)
    {
      if(ret[j] == currentKey)
      {
        alreadyInRet = true;
        break;
      }
    }
    if(alreadyInRet === false)
    {
      ret[retLen] = currentKey;
    }
  }
  return ret;
}
